/*NOT EXISTSを使って欠番を求める*/
SELECT
	sv.seq
FROM
	Sequence AS sv
WHERE
	NOT EXISTS
	(
		SELECT
			seq
		FROM
			SeqTbl AS s
		WHERE
			s.seq = sv.seq
	)
;

/*外部結合を使って欠番を求める*/
SELECT
	sv.seq
FROM
	Sequence AS sv
LEFT JOIN
	SeqTbl AS s
ON
	s.seq = sv.seq
WHERE
	s.seq IS NULL
;

|   2 |

|   5 |

|   7 |
|   8 |
|  11 |
|  12 |
