/*取得するデータを指定*/
SELECT
	/*ナンバー(教本のカラム名はkeyだが、生成時にエラーが出てしまう
	ため、'number'で代用)カラムをkeyと命名して指定*/
	number AS 'key'
/*テーブルを指定*/
FROM
	/*Greatests(教本準拠)テーブルを指定*/
	Greatests
/*データを指定した順にソート*/
ORDER BY
	/*'C'のデータの順番を指定*/
	number = 'C',
	/*'D'のデータの順番を指定*/
	number = 'D',
	/*'A'のデータの順番を指定*/
	number = 'A',
	/*'B'のデータの順番を指定*/
	number = 'B'
;

| key |
+-----+
| B   |
| A   |
| D   |
| C   |
+-----+
