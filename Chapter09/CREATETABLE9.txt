	Skills

DROP TABLE
	EmpSkills
;

/*使用するテーブルを作成*/
CREATE TABLE
	Skills
	(
		skill VARCHAR(32),
		PRIMARY KEY(skill)
	)
;

/*使用するテーブルを作成*/
CREATE TABLE
	EmpSkills
	(
		emp VARCHAR(32),
		skill VARCHAR(32),
		PRIMARY KEY(emp, skill)
	)
;
